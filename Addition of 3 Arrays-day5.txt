
#include <stdio.h>
int main() {
    int size = 5;
    int a[] = {1, 2, 3, 4, 5};
    int b[] = {10, 20, 30, 40, 50};
    int c[] = {100, 200, 300, 400, 500};
    int result[size];
    for (int i = 0; i < size; i++) {
        result[i] = a[i] + b[i] + c[i];
    }
    printf("Resultant array after adding:\n");
    for (int i = 0; i < size; i++) {
        printf("%d ", result[i]);
    }
    return 0;
}
