#include <stdio.h>
int main() {
    int size, i;
    printf("Enter the size of the arrays: ");
    scanf("%d", &size);
    int a[size], b[size], c[size], result[size];
    printf("Enter elements of array A:\n");
    for(i = 0; i < size; i++) {
        scanf("%d", &a[i]);
    }
    printf("Enter elements of array B:\n");
    for(i = 0; i < size; i++) {
        scanf("%d", &b[i]);
    }
    printf("Enter elements of array C:\n");
    for(i = 0; i < size; i++) {
        scanf("%d", &c[i]);
    }
    for(i = 0; i < size; i++) {
        result[i] = a[i] * b[i] * c[i];
    }
    printf("Result of element-wise multiplication:\n");
    for(i = 0; i < size; i++) {
        printf("%d ", result[i]);
    }
    return 0;
}
